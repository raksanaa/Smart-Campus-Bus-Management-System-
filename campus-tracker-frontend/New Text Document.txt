        function showMessage(message, type = 'success') {
            const box = document.getElementById('message-box');
            box.textContent = message;
            box.classList.remove('hidden', 'bg-green-600', 'bg-red-600', 'bg-yellow-600');
            
            if (type === 'success') box.classList.add('bg-green-600');
            else if (type === 'error') box.classList.add('bg-red-600');
            else box.classList.add('bg-yellow-600');

            box.classList.remove('hidden');

            setTimeout(() => {
                box.classList.add('hidden');
            }, 5000);
        }

        function setView(view, tripId = null) {
            currentView = view;
            selectedTripId = tripId;

            document.querySelectorAll('.tab-button').forEach(btn => btn.classList.remove('active'));
            document.getElementById(`tab-${view.split('-')[0]}`).classList.add('active');
            
            if (view === 'active') fetchActiveTrips();
            else if (view === 'logs' && tripId) fetchTripLogs(tripId);
            else if (view === 'shuttles') fetchShuttles();
            else if (view === 'users') fetchUsers();

        // --- API HELPERS ---

        async function fetchData(endpoint) {
            try {
                const response = await fetch(`${API_BASE_URL}/${endpoint}`);
                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                return await response.json();
            } catch (error) {
                showMessage(`Error fetching ${endpoint}: ${error.message}. Check CORS/Backend.`, 'error');
                return null;
            }
        }

        async function postOrPutData(endpoint, data, method = 'POST') {
            try {
                const response = await fetch(`${API_BASE_URL}/${endpoint}`, {
                    method: method,
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(data),
                });
                
                if (response.status === 201 || response.status === 200) {
                    showMessage(`${method === 'POST' ? 'Created' : 'Updated'} successfully!`, 'success');
                    return await response.json();
                } else {
                    const errorText = await response.text();
                    throw new Error(`Failed: ${errorText}`);
                }
            } catch (error) {
                showMessage(`Submission Error: ${error.message}`, 'error');
                return null;
            }
        }

        async function handleLogSubmission(e, tripId) {
            e.preventDefault();
            
            const action = document.getElementById('action').value;
            const userIdInput = document.getElementById('userId').value;
            const remarks = document.getElementById('remarks').value;

            const payload = {
                action: action,
                remarks: remarks,
            };
            
            // Only include the user field if a userId was provided
            if (userIdInput) {
                payload.user = { userId: parseInt(userIdInput) };
            }

            const savedLog = await postOrPutData(`trips/${tripId}/logs`, payload, 'POST');
            
            if (savedLog) {
                fetchTripLogs(tripId); // Refresh the logs view
            }
        }

### Key Improvements and Features

1.  **Attractiveness:** Uses a more appealing dark-mode aesthetic with the **Poppins font** and a **vibrant purple accent color** (`--primary`).
2.  **Navigation:** Added new tabs for **Shuttle Management** and **User Management**.
3.  **New Components:** Each new tab includes:
    * A card for **adding new entities** (User/Shuttle) with a basic form placeholder.
    * A card to **list all existing entities** fetched from your API.
4.  **Error Handling:** The `fetchShuttles()` and `fetchUsers()` functions currently use generic endpoints (`/shuttles` and `/users`). **You will need to ensure your Spring Boot controllers have these endpoints** to see data on those tabs.
5.  **Re-use:** The successful logic for fetching active trips and managing logs remains fully functional!

**Action:** Replace the old `index.html` file, refresh your browser, and enjoy your new, professional-looking dashboard!
